3.  SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME,A.ACCOUNT_NO, A.ACCOUNT_TYPE, B.BANK_NAME FROM CUSTOMER_PERSONAL_INFO C, BANK_INFO B, ACCOUNT_INFO A WHERE B.IFSC_CODE=A.IFSC_CODE AND A.CUSTOMER_ID=C.CUSTOMER_ID AND A.ACCOUNT_NO IS NOT NULL;

4.  SELECT  CUSTOMER_ID, CUSTOMER_NAME,GENDER,MARITAL_STATUS, CUSTOMER_NAME||'_'||GENDER||'_'||MARITAL_STATUS AS UNIQUE_REF_STRING FROM CUSTOMER_PERSONAL_INFO;

5.  SELECT ACCOUNT_NO, CUSTOMER_ID, REGISTRATION_DATE, INITIAL_DEPOSIT FROM ACCOUNT_INFO WHERE INITIAL_DEPOSIT BETWEEN 15000 AND 25000;

6. SELECT CUSTOMER_ID, CUSTOMER_NAME, DATE_OF_BIRTH, GUARDIAN_NAME FROM CUSTOMER_PERSONAL_INFO WHERE CUSTOMER_NAME LIKE 'J%';

7. SELECT CUSTOMER_ID, ACCOUNT_NO, (SUBSTR(CUSTOMER_ID,-3)||SUBSTR(ACCOUNT_NO,-4)) AS PASSCODE FROM ACCOUNT_INFO;

8. select customer_id, customer_name, date_of_birth, marital_status, gender, guardian_name,contact_no, mail_id from customer_personal_info where gender='M' and marital_status='MARRIED';

9. SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, C.GUARDIAN_NAME, R.REFERENCE_ACC_NAME AS FRIEND FROM CUSTOMER_PERSONAL_INFO C, CUSTOMER_REFERENCE_INFO R WHERE C.CUSTOMER_ID=R.CUSTOMER_ID;

10. SELECT CUSTOMER_ID, ACCOUNT_NO, ('$'||ROUND(INTEREST,0)) AS INTEREST_AMT FROM ACCOUNT_INFO;

11. SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, A.ACCOUNT_TYPE, A.ACTIVATION_DATE, B.BANK_NAME FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A, BANK_INFO B WHERE A.ACTIVATION_DATE='10-APR-2012' AND A.CUSTOMER_ID=C.CUSTOMER_ID AND B.IFSC_CODE=A.IFSC_CODE;

12. SELECT A.ACCOUNT_NO, C.CUSTOMER_ID, C.CUSTOMER_NAME, B.BANK_NAME, B.BRANCH_NAME, B.IFSC_CODE, C.CITIZENSHIP, A.INTEREST, A.INITIAL_DEPOSIT FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A, BANK_INFO B WHERE C.CUSTOMER_ID=A.CUSTOMER_ID AND B.IFSC_CODE=A.IFSC_CODE;

13.  SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, C.DATE_OF_BIRTH, C.GUARDIAN_NAME, C.CONTACT_NO, C.MAIL_ID, R.REFERENCE_ACC_NAME FROM CUSTOMER_PERSONAL_INFO C, CUSTOMER_REFERENCE_INFO R WHERE C.IDENTIFICATION_DOC_TYPE='PASSPORT' AND R.CUSTOMER_ID=C.CUSTOMER_ID;

14. SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, A.ACCOUNT_TYPE, A.INITIAL_DEPOSIT FROM ACCOUNT_INFO A, CUSTOMER_PERSONAL_INFO C WHERE A.CUSTOMER_ID=C.CUSTOMER_ID AND A.INITIAL_DEPOSIT=(SELECT MAX(INITIAL_DEPOSIT) FROM ACCOUNT_INFO);

15.  SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, A.ACCOUNT_TYPE, A.INTEREST, B.BANK_NAME, A.INITIAL_DEPOSIT FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A, BANK_INFO B WHERE B.IFSC_CODE=A.IFSC_CODE AND C.CUSTOMER_ID=A.CUSTOMER_ID AND A.INTEREST=(SELECT MAX(INTEREST) FROM ACCOUNT_INFO);

16.  SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, B.BANK_NAME, C.CONTACT_NO, C.MAIL_ID FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A, BANK_INFO B WHERE C.CUSTOMER_ID=A.CUSTOMER_ID AND B.IFSC_CODE=A.IFSC_CODE AND C.ADDRESS LIKE '%BANGALORE';

17.  SELECT C.CUSTOMER_ID, B.BANK_NAME, B.BRANCH_NAME, B.IFSC_CODE, A.REGISTRATION_DATE, A.ACTIVATION_DATE FROM CUSTOMER_PERSONAL_INFO C INNER JOIN ACCOUNT_INFO A ON C.CUSTOMER_ID=A.CUSTOMER_ID INNER JOIN BANK_INFO B ON B.IFSC_CODE=A.IFSC_CODE WHERE to_char(AI.ACTIVATION_DATE, 'Mon')='Mar';

18.  SELECT ((INTEREST/100)*INITIAL_DEPOSIT) AS INTEREST_AMT, C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, A.ACCOUNT_TYPE, A.INTEREST, A.INITIAL_DEPOSIT FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A WHERE C.CUSTOMER_ID=A.CUSTOMER_ID;

19.  SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, C.DATE_OF_BIRTH, C.GUARDIAN_NAME, C.CONTACT_NO, C.MAIL_ID, R.REFERENCE_ACC_NAME FROM CUSTOMER_PERSONAL_INFO C, CUSTOMER_REFERENCE_INFO R WHERE C.CUSTOMER_ID=R.CUSTOMER_ID AND R.REFERENCE_ACC_NAME='RAGHUL';

20. SELECT CUSTOMER_ID, CUSTOMER_NAME,('+91-'||SUBSTR(CONTACT_NO,1,3)||'-'||SUBSTR(CONTACT_NO,4,3)||'-'||SUBSTR(CONTACT_NO,-4)) AS CONTACT_ISD FROM CUSTOMER_PERSONAL_INFO ORDER BY CUSTOMER_ID DESC;

21. SELECT A.ACCOUNT_NO, A.ACCOUNT_TYPE, C.CUSTOMER_ID, C.CUSTOMER_NAME, C.DATE_OF_BIRTH, C.GUARDIAN_NAME, C.CONTACT_NO, C.MAIL_ID, C.GENDER,
R.REFERENCE_ACC_NAME, R.REFERENCE_ACC_NO, A.REGISTRATION_DATE, A.ACTIVATION_DATE, ACTIVATION_DATE-REGISTRATION_DATE AS NoofdaysforActivation,
B.BANK_NAME, B.BRANCH_NAME, A.INITIAL_DEPOSIT
FROM CUSTOMER_PERSONAL_INFO C, BANK_INFO B, CUSTOMER_REFERENCE_INFO R, ACCOUNT_INFO A WHERE C.CUSTOMER_ID=A.CUSTOMER_ID AND R.CUSTOMER_ID=C.CUSTOMER_ID
AND B.IFSC_CODE=A.IFSC_CODE;

22. SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, C.GUARDIAN_NAME, C.IDENTIFICATION_DOC_TYPE, R.REFERENCE_ACC_NAME, A.ACCOUNT_TYPE, B.IFSC_CODE, B.BANK_NAME, TRUNC((INITIAL_DEPOSIT+INTEREST),0) AS CURRENT_BALANCE FROM CUSTOMER_PERSONAL_INFO C, BANK_INFO B, ACCOUNT_INFO A, CUSTOMER_REFERENCE_INFO R WHERE C.CUSTOMER_ID=A.CUSTOMER_ID AND R.CUSTOMER_ID=C.CUSTOMER_ID AND R.CUSTOMER_ID=C.CUSTOMER_ID AND ACCOUNT_TYPE='SAVINGS';

23. SELECT C.CUSTOMER_ID, C.CUSTOMER_NAME, A.ACCOUNT_NO, A.ACCOUNT_TYPE, A.INTEREST, A.INITIAL_DEPOSIT,
CASE
WHEN A.INITIAL_DEPOSIT=20000 THEN 'high'
WHEN A.INITIAL_DEPOSIT=16000 THEN 'moderate'
WHEN A.INITIAL_DEPOSIT=10000 THEN 'average'
WHEN A.INITIAL_DEPOSIT=5000 THEN 'low'
WHEN A.INITIAL_DEPOSIT=0 THEN 'very low'
ELSE 'invalid' END DEPOSIT_STATUS
FROM CUSTOMER_PERSONAL_INFO C, ACCOUNT_INFO A
WHERE C.CUSTOMER_ID=A.CUSTOMER_ID;
